\begin{align}
e^{2 \pi i (N-k) \frac{n}{N}} &= e^{2 \pi i n \frac{(N-k)}{N}} \\
&= e^{2\pi i n \left( 1 - \frac{k}{N} \right)} \\
&= e^{ 2\pi i n - 2\pi i n\frac{k}{N} } \\
&= e^{ 2\pi i n} \cdot e^{- 2\pi i n\frac{k}{N} } \\
&= 1 \cdot e^{- 2\pi i n\frac{k}{N} } & \text{(n is integer)} \\
&= e^{- 2\pi i k\frac{n}{N} } \\
\end{align}
